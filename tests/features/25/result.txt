=== C TEST CASE ===

volatile int arr1[100];
volatile int arr2[100];

int sumarray(void) {
    int sum = 0;
    for (int i = 0; i < 100; ++i)
        sum += arr1[i] + arr2[i];
    return sum;
}

volatile int data[50];
int singlesum(void) {
    int total = 0;
    for (int i = 0; i < 50; ++i)
        total += data[i];
    return total;
}

int main(void) {
    return sumarray() + singlesum();
}
