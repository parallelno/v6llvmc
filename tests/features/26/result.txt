=== O44 Adjacent XCHG Cancellation — Feature Test 26 ===

--- C Test Case ---

volatile int arr1[100];
volatile int arr2[100];

int sumarray(void) {
    int sum = 0;
    for (int i = 0; i < 100; ++i)
        sum += arr1[i] + arr2[i];
    return sum;
}

int main(void) {
    return sumarray();
}

--- c8080 ASM (sumarray loop body, converted to i8080 mnemonics) ---

sumarray:
    LXI  HL, 0
    SHLD sumarray_sum          ; sum = 0
    SHLD sumarray_i            ; i = 0
l_0:
    LHLD sumarray_i            ; load i
    XCHG                       ; DE = i
    LXI  HL, 100               ; HL = 100
    ; __o_sub_16 inlined:
    MOV  A, L
    SUB  E
    MOV  L, A
    MOV  A, H
    SBB  D
    MOV  H, A
    JP   l_2                   ; if i >= 100, exit

    LXI  DE, arr1              ; DE = &arr1[0]
    LHLD sumarray_i            ; HL = i
    DAD  HL                    ; HL = i*2
    DAD  DE                    ; HL = &arr1[i]
    MOV  E, M                  ; load arr1[i] lo
    INX  HL
    MOV  D, M                  ; load arr1[i] hi
    PUSH DE                    ; save arr1[i]
    LXI  DE, arr2              ; DE = &arr2[0]
    LHLD sumarray_i            ; HL = i
    DAD  HL                    ; HL = i*2
    DAD  DE                    ; HL = &arr2[i]
    MOV  E, M                  ; load arr2[i] lo
    INX  HL
    MOV  D, M                  ; load arr2[i] hi
    XCHG                       ; HL = arr2[i]
    POP  DE                    ; DE = arr1[i]
    DAD  DE                    ; HL = arr1[i] + arr2[i]
    XCHG                       ; DE = sum of elements
    LHLD sumarray_sum          ; HL = sum
    DAD  DE                    ; HL = sum + arr1[i] + arr2[i]
    SHLD sumarray_sum          ; store sum
    LHLD sumarray_i            ; HL = i
    INX  HL                    ; i++
    SHLD sumarray_i            ; store i
    JMP  l_0                   ; loop
l_2:
    LHLD sumarray_sum          ; return sum
    RET
