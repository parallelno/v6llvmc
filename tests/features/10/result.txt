== C Test Case ==

extern int bar(int x);

int test_ne_zero(int x) {
    if (x) return bar(x);
    return 0;
}

int test_eq_zero(int x) {
    if (!x) return 0;
    return bar(x);
}

int test_multi_cond(int x) {
    if (x == 1) return 10;
    if (x == 2) return 20;
    return bar(x);
}

int test_null_guard(int *p) {
    if (p) return *p;
    return -1;
}
