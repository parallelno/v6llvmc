================================================================================
O54: Optimal stack adjustment via PUSH/POP for small frames
================================================================================

Summary
-------
For SP adjustments with small even sizes (n ∈ {2,4} in non-Size mode,
n ∈ {2,4,6} in Size mode), the V6C backend now emits PUSH/POP of a
register pair whose halves are dead at the adjustment point. When no
dead pair is available, it falls back to DCX/INX SP × n for n ∈ {2,4}
(no clobber, 1B/8cc per step). Otherwise it keeps the existing
LXI+DAD+SPHL sequence (5B/32cc, clobbers HL+FLAGS).

Costs:
  PUSH r           : 1 B / 16 cc   (SP -= 2)
  POP  r           : 1 B / 12 cc   (SP += 2)
  DCX/INX SP       : 1 B /  8 cc   (SP -= 1 / SP += 1)
  LXI+DAD+SPHL     : 5 B / 32 cc   (clobbers HL, FLAGS)

For a 2-byte even frame: PUSH+POP = 2 B / 28 cc beats LXI×2 = 10 B / 64 cc.
For a 4-byte even frame: 2× PUSH/POP = 4 B / 56 cc beats LXI×2 = 10 B / 64 cc.

================================================================================
C test case (tests/features/47/v6llvmc.c)
================================================================================
extern unsigned char ext_fn(unsigned char x, unsigned char y);

volatile unsigned int g_sink;

unsigned int worker(unsigned char a, unsigned char b, unsigned char c) {
    unsigned char r1 = ext_fn(a, b);
    unsigned char r2 = ext_fn(b, c);
    unsigned char r3 = ext_fn(a, c);
    return (unsigned int)r1 + r2 + r3;
}

int main(int argc, char **argv) {
    (void)argc; (void)argv;
    g_sink = worker(1, 2, 3);
    return 0;
}

    MVI    A, 3
    CALL   worker
    SHLD   g_sink
    LXI    H, 0
    RET

worker:
    STA    __a_3_worker
    LDA    __a_1_worker
    STA    __a_1_ext_fn
    LDA    __a_2_worker
    CALL   ext_fn
    STA    worker_r1
    LDA    __a_2_worker
    STA    __a_1_ext_fn
    LDA    __a_3_worker
    CALL   ext_fn
    STA    worker_r2
    LDA    __a_1_worker
    STA    __a_1_ext_fn
    LDA    __a_3_worker
    CALL   ext_fn
    STA    worker_r3
    LHLD   worker_r1
    MVI    H, 0
    XCHG
    LHLD   worker_r2
    MVI    H, 0
    DAD    D
    XCHG
    LHLD   worker_r3
    MVI    H, 0
    DAD    D
    RET
