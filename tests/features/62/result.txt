extern unsigned char op1(unsigned char);
extern void          sink(unsigned char);
extern void          use2(unsigned char a, unsigned char b);

unsigned char shape_a(unsigned char a) {
    if (a) return 1;
    return 0;
}

unsigned char shape_a_dead(unsigned char x, unsigned char y) {
    if (y) return x;
    return 0;
}

unsigned char shape_a_live(unsigned char val_in_A, unsigned char cond) {
    unsigned char r = op1(val_in_A);
    if (cond) return (unsigned char)(r + 1);
    return r;
}

unsigned char shape_a_live_loop(unsigned char seed, unsigned char n) {
    unsigned char acc = seed;
    while (n) {
        acc = op1(acc);
        n--;
    }
    return acc;
}

int main(void) {
    use2(shape_a(0x11),            shape_a(0));
    use2(shape_a_dead(0x22, 0x33), shape_a_dead(0x44, 0));
    use2(shape_a_live(0x55, 1),    shape_a_live(0x66, 0));
    sink(shape_a_live_loop(0x77, 5));
    return 0;
}

NOTES:
- shape_a (shape 1) is unchanged. Already optimal pre-O80.
- shape_a_dead's "A dead" expectation didn't hold once the function
  body returned %x — A is live at the test, so the post-RA expander
  selected shape 3 (INR/DCR). This is a strict win versus the old
  scratch-save/restore sequence.
- shape_a_live's compare path is now 2B / 16cc (INR + DCR) instead of
  4B / 24cc (MOV + XRA + CMP + MOV-restore). The +1B in total size
  comes from an unrelated downstream change in the deferred-zero
  spill setup (MVI L,0 vs MOV L,A on the cold path), not the compare.
- shape_a_live_loop is the headline win: −24cc per loop iteration
  and the H/L scratch is freed for other use across the test.
- All 137 lit tests pass; 5 golden tests pass.
