=== O43 SHLD/LHLD to PUSH/POP Peephole — Feature Test 27 ===

--- C Test Case ---

volatile int arr1[200];
volatile int arr2[200];

int sumarray(void) {
    int sum = 0;
    for (int i = 0; i < 200; ++i)
        sum += arr1[i] + arr2[i];
    return sum;
}

int main(void) {
    return sumarray();
}

--- c8080 ASM (sumarray, converted to i8080 mnemonics) ---

sumarray:
    LXI  HL, 0
    SHLD sumarray_sum          ; sum = 0
    SHLD sumarray_i            ; i = 0
l_0:
    LXI  DE, 200               ; compare limit
    CALL __o_sub_16            ; HL = i - 200
    JP   l_2                   ; if i >= 200, exit

    LXI  DE, arr1              ; &arr1[0]
    LHLD sumarray_i            ; HL = i
    DAD  HL                    ; HL = i*2
    DAD  DE                    ; HL = &arr1[i]
    MOV  E, M                  ; load arr1[i] lo
    INX  HL
    MOV  D, M                  ; load arr1[i] hi
    PUSH DE                    ; save arr1[i]
    LXI  DE, arr2              ; &arr2[0]
    LHLD sumarray_i            ; HL = i
    DAD  HL                    ; HL = i*2
    DAD  DE                    ; HL = &arr2[i]
    MOV  E, M                  ; load arr2[i] lo
    INX  HL
    MOV  D, M                  ; load arr2[i] hi
    XCHG                       ; HL = arr2[i]
    POP  DE                    ; DE = arr1[i]
    DAD  DE                    ; HL = arr1[i] + arr2[i]
    XCHG                       ; DE = element sum
    LHLD sumarray_sum          ; HL = sum
    DAD  DE                    ; HL += element sum
    SHLD sumarray_sum          ; store sum
    LHLD sumarray_i            ; HL = i
    INX  HL                    ; i++
    SHLD sumarray_i            ; store i
    JMP  l_0                   ; loop
l_2:
    LHLD sumarray_sum          ; return sum
    RET

__o_sub_16:
    MOV  A, L
    SUB  E
    MOV  L, A
    MOV  A, H
    SBB  D
    MOV  H, A
    RET

--- v6llvmc ASM (sumarray with O43) ---

--- O43 improvement (vs same regalloc without peephole) ---

Without O43, the loop has SHLD/LHLD instead of PUSH/POP:
  SHLD __v6c_ss.sumarray+4  20cc 3B  →  PUSH HL  16cc 1B  (saves 4cc 2B)
  SHLD __v6c_ss.sumarray+2  20cc 3B  →  PUSH HL  16cc 1B  (saves 4cc 2B)
  LHLD __v6c_ss.sumarray+2  20cc 3B  →  POP  HL  12cc 1B  (saves 8cc 2B)
  LHLD __v6c_ss.sumarray+4  20cc 3B  →  POP  HL  12cc 1B  (saves 8cc 2B)

  Savings per iteration:  24cc, 8B
  Savings (200 iter):     4,800cc total
  Before O43:             69,696cc, 49B loop (69B function)
  After  O43:             64,896cc, 41B loop (61B function)
  Improvement:            6.9% faster, 8B smaller loop body

  LHLD __v6c_ss.sumarray (arr2_ptr) correctly NOT converted:
    intervening PUSH/POP between SHLD and LHLD makes SPDelta != 0 → abort.
