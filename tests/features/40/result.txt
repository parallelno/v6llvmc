O61 Stage 6 — non-A i8 spill sources — feature test 40
=======================================================

## C source (tests/features/40/v6llvmc.c, same for c8080.c aside from
## extern attributes which c8080 doesn't accept)

    __attribute__((leaf)) extern unsigned char op1(unsigned char x);
    __attribute__((leaf)) extern unsigned char op2(unsigned char x);
    __attribute__((leaf)) extern void use3(unsigned char a, unsigned char b,
                                           unsigned char c);
    __attribute__((leaf)) extern void use4(unsigned char a, unsigned char b,
                                           unsigned char c, unsigned char d);

    unsigned char three_i8(unsigned char x, unsigned char y, unsigned char z) {
        unsigned char a = op1(x);
        unsigned char b = op2(y);
        unsigned char c = op1(z);
        use3(a, b, c);
        return (unsigned char)(a + b + c);
    }

    unsigned char four_i8(unsigned char x, unsigned char y,
                          unsigned char z, unsigned char w) {
        unsigned char a = op1(x);
        unsigned char b = op2(y);
        unsigned char c = op1(z);
        unsigned char d = op2(w);
        use4(a, b, c, d);
        return (unsigned char)(a + b + c + d);
    }

    /* main() exercises three_i8 + four_i8 end-to-end. */

## v6llvmc assembly (Stage 5 baseline — v6llvmc.asm)
## =================================================
## Pre-Stage-6, the A-source spills already route through O61 (patched
## STA .LLo61_N+1 + MVI target, 0). Non-A i8 spills (B/C/D/E/H/L)
## fall through to the classical O64 ladder against BSS
## __v6c_ss.three_i8 / __v6c_ss.four_i8 / __v6c_ss.main.
##
## three_i8 (Stage 5): 28 insns, 14 BSS refs to __v6c_ss.three_i8
##                     + 2 patched LLo61 refs. 3 bytes BSS reserved.
## four_i8  (Stage 5): 40 insns, 11 BSS refs to __v6c_ss.four_i8
##                     + 4 patched LLo61 refs. 4 bytes BSS reserved.
## main     (Stage 5): 57 insns, 3 BSS refs to __v6c_ss.main
##                     + 16 patched LLo61 refs. 7 bytes BSS reserved.

## v6llvmc assembly (Stage 6 — v6llvmc_new02.asm)
## ==============================================
## Every non-A i8 spill in three_i8/four_i8 is now routed through
## O64's expandSpill8Static with a .LLo61_N+1 appender, terminating
## in `LXI HL, .LLo61_N+1; MOV M, r` (Row 1, HL dead at the spill).
## The reload becomes `MVI r, 0` at the `.LLo61_N:` label — the imm
## byte was written by the spill. Classical BSS refs for these slots
## are eliminated entirely.

    three_i8:
        LXI  HL, .LLo61_1+1      ; spill C (Row 1, HL dead)
        MOV  M, C
        LXI  HL, .LLo61_2+1      ; spill E (Row 1, HL dead)
        MOV  M, E
        CALL op1
        STA  .LLo61_0+1          ; spill A (Stage 4 fast path)
        LDA  .LLo61_2+1
        CALL op2
        STA  .LLo61_2+1
        LDA  .LLo61_1+1
        CALL op1
        MOV  C, A
        LXI  HL, .LLo61_1+1
        MOV  M, C
        LDA  .LLo61_0+1
    .LLo61_2:
        MVI  E, 0
        CALL use3
    .LLo61_0:
        MVI  L, 0
        LDA  .LLo61_2+1
        ADD  L
    .LLo61_1:
        MVI  L, 0
        ADD  L
        RET

## Bytes / cycles
## -----------------
## For three_i8 non-A slots (C and E), each saves (vs classical
## O64 Row 1 reload `LXI HL, __v6c_ss.three_i8+N; MOV r, M` → 20 cc,
## 4 B) by collapsing to `MVI r, 0` (7 cc, 2 B):
##   Δ per slot: −13 cc, −2 B code + −1 B BSS (dead).
## Δ three_i8 overall: ~ −26 cc, −4 B across two non-A reloads.
## Δ four_i8 overall:  ~ −39 cc, −6 B across three non-A reloads.

## Bug fixed as part of Stage 6
## ----------------------------
## V6CLoadImmCombine treated `MVI r, imm` carrying MO_PATCH_IMM as a
## trackable constant def. With two patched MVIs landing the same
## (reg, imm) pair, the "same value already in reg → erase MVI"
## optimisation erased the second patched MVI and corrupted the
## reload. Fixed by gating all four immediate-handling sites in
## V6CLoadImmCombine.cpp on a new `isPlainImm(MO)` helper that also
## requires `MO.getTargetFlags() == 0`. Any operand carrying a
## non-zero target flag (MO_PATCH_IMM in practice) is treated as an
## opaque constant def (invalidate dst, do not track). This also
## fixes a latent Stage 4 bug.

## Dead BSS elimination
## ---------------------
## After O61 rewrites spill/reload pseudos to patched-reload-site
## addresses, V6CSpillPatchedReload scans the function for surviving
## FrameIndex operands and drops any static-stack slot that is no
## longer referenced. Surviving slots are repacked monotonically and
## the backing `@__v6c_ss.<func>` global is resized (or erased
## entirely when every slot was patched away), so AsmPrinter emits a
## smaller `.comm` — or none at all. In this test, all three
## `.comm __v6c_ss.three_i8,3,1`, `.comm __v6c_ss.four_i8,4,1`, and
## `.comm __v6c_ss.main,7,1` directives are eliminated: net BSS
## saved = 3 + 4 + 7 = 14 bytes.
##
## Compaction is a fixed-point post-rewrite step that is agnostic to
## slot width (i8 / i16) and agnostic to patching origin — any FI
## that is no longer referenced by any MachineOperand in the function
## is dropped.
